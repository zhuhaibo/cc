    ajax

    <script type="text/javascript">
	$(function(){
		$(".ajaxInputBtn").click(function(){
			var val = $(".ajaxInput").val();
			if(val != ""){
				$.baseAjax({
					url:"test.php",
					data: {"value":val}
				},function(msg){
					console.log(msg)
					if(msg.error_code){
						$("#inputTxt").html("正确");
					}else {
						$("#inputTxt").html(msg.error_msg.value);
					}
				});
			}
		});
	});
</script>
